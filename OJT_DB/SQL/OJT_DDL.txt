CREATE TABLE EMPLOYEE
(
	EMPLOYEE_ID int int NOT NULL AUTO_INCREMENT COMMENT  '従業員ID',
	DEPARTMENT_ID int int NOT NULL COMMENT  '部署ID',
	NAME varchar(100) NOT NULL COMMENT '名前',
	AGE int NOT NULL NOT NULL COMMENT '年齢',
	WORKFORM_CODE int NOT NULL COMMENT '勤務形態区分',
	REG_DT datetime NOT NULL COMMENT '登録日時',
	REG_NAME varchar(128) NOT NULL COMMENT '登録者',
	UPD_DT datetime NOT NULL COMMENT '更新日時',
	UPD_NAME varchar(128) NOT NULL COMMENT '更新者',
	VERSION int NOT NULL COMMENT 'バージョン',
	PRIMARY KEY (EMPLOYEE_ID)
) COMMENT = '従業員';

CREATE TABLE ATTENDANCE_MANAGEMENT
(
	ATTENDANCE_MANAGEMENT_ID int int NOT NULL AUTO_INCREMENT COMMENT  '勤怠ID',
	EMPLOYEE_ID int int NOT NULL COMMENT  '従業員ID',
	WORKING_START_TIME datetime NOT NULL COMMENT '出勤時間',
	WORKING_END_TIME datetime NOT NULL NOT NULL COMMENT '退勤時間',
	REG_DT datetime NOT NULL COMMENT '登録日時',
	REG_NAME varchar(128) NOT NULL COMMENT '登録者',
	UPD_DT datetime NOT NULL COMMENT '更新日時',
	UPD_NAME varchar(128) NOT NULL COMMENT '更新者',
	VERSION int NOT NULL COMMENT 'バージョン',
	PRIMARY KEY (EMPLOYEE_ID)
) COMMENT = '勤怠';

CREATE TABLE DEPARTMENT
(
	DEPARTMENT_ID int int NOT NULL AUTO_INCREMENT COMMENT  '部署ID',
	DEPARTMENT_NAME varchar(100) int NOT NULL COMMENT  '部署名',
	REG_DT datetime NOT NULL COMMENT '登録日時',
	REG_NAME varchar(128) NOT NULL COMMENT '登録者',
	UPD_DT datetime NOT NULL COMMENT '更新日時',
	UPD_NAME varchar(128) NOT NULL COMMENT '更新者',
	VERSION int NOT NULL COMMENT 'バージョン',
	PRIMARY KEY (DEPARTMENT_ID)
) COMMENT = '部署マスタ';

CREATE TABLE WORKFORM
(
	WORKFORM_ID int int NOT NULL AUTO_INCREMENT COMMENT  '部署ID',
	WORKFORM_NAME varchar(100) int NOT NULL COMMENT  '勤務形態名',
	REG_DT datetime NOT NULL COMMENT '登録日時',
	REG_NAME varchar(128) NOT NULL COMMENT '登録者',
	UPD_DT datetime NOT NULL COMMENT '更新日時',
	UPD_NAME varchar(128) NOT NULL COMMENT '更新者',
	VERSION int NOT NULL COMMENT 'バージョン',
	PRIMARY KEY (DEPARTMENT_ID)
) COMMENT = '勤務形態マスタ';
